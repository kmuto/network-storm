# dummyモジュールのロード
sudo modprobe dummy
   
# インターフェイス 'dum0' を作成
sudo ip link add dum0 type dummy
sudo ip link set dum0 up
   
# 10.0.0.0/22 (1024個のIP) への通信を dum0 に向ける
# これにより、この範囲のIPへのpingは全てdum0に吸い込まれます
sudo ip route add 10.0.0.0/22 dev dum0

sudo sysctl -w net.ipv4.conf.all.rp_filter=0
sudo sysctl -w net.ipv4.conf.dum0.rp_filter=0
# 念のため default も
sudo sysctl -w net.ipv4.conf.default.rp_filter=0
